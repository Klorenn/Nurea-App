import { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  const baseUrl = process.env.NEXT_PUBLIC_APP_URL || 'https://nurea.app'

  return {
    rules: [
      {
        userAgent: '*',
        allow: ['/', '/explore', '/professionals/'],
        disallow: [
          '/dashboard/',
          '/admin/',
          '/api/',
          '/checkout/',
          '/auth/reset-password',
          '/consultation/',
        ],
      },
    ],
    sitemap: `${baseUrl}/sitemap.xml`,
  }
}
